2.9 Windspeed Analysis

ws <- read.csv("2-Clean Data/windspeeds.csv") %>% mutate(date = ymd(date))
diffusion_ws <- diffusion%>% mutate(date = date(start_time)) %>% 
  left_join(ws, by = c("site", "location", "date","trip")) %>% 
  filter(is.na(ws_mean)==F) %>% 
  mutate(sampling = case_when(
    date == "2022-08-16" ~ "Deans1",
    date == "2022-08-17" ~ "Deans1",
    date == "2022-08-30" ~ "Deans2",
    date == "2022-08-31" ~ "Deans2",
    date == "2022-09-06" ~ "Catfish1",
    date == "2022-09-07" ~ "Catfish1",
    date == "2022-09-18" ~ "Catfish2",
    date == "2022-09-19" ~ "Catfish2",
    date == "2022-09-13" ~ "Pick1",
    date == "2022-09-14" ~ "Pick1",
    date == "2022-08-22" ~ "Sister31",
    date == "2022-08-23" ~ "Sister31"
  ))

ch4_ws <- lmer(flux_ch4_g_hr ~ ws_mean + (1|sampling), data = diffusion_ws)
summary(ch4_ws)
